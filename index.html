// Reverse a portion of a string
function reverseString(st, a, b) {
    let chars = st.split("");
    while (a < b) {
        const temp = chars[a];
        chars[a] = chars[b];
        chars[b] = temp;

        a++;
        b--;
    }
    return chars.join("");
}
